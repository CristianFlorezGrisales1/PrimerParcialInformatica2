//funciones
#include <iostream>
using namespace std;

void dibujarPatrones() {
    int dim = 8, posicion1 = 3, posicion2 = 4;

    // Patr贸n 1
    cout << "Patron 1:" << endl;
    bool matriz1[8][8];

    for (int i = 0; i < dim; i++) {
        if (i <= 4) {
            for (int j = 0; j < dim; j++) {
                if (j >= posicion1 && j <= posicion2) {
                    matriz1[i][j] = true;
                } else {
                    matriz1[i][j] = false;
                }
            }
            posicion1--;
            posicion2++;
        }

        if (i >= 5) {
            if (i == 5) {
                posicion1 = 1;
                posicion2 = 6;
            }

            if (i == 6) {
                posicion1 = 2;
                posicion2 = 5;
            }

            if (i == 7) {
                posicion1 = 3;
                posicion2 = 4;
            }

            for (int j = 0; j < dim; j++) {
                if (j >= posicion1 && j <= posicion2) {
                    matriz1[i][j] = true;
                } else {
                    matriz1[i][j] = false;
                }
            }
        }
    }

    for (int i = 0; i < dim; i++) {
        for (int j = 0; j < dim; j++) {
            cout << matriz1[i][j];
        }
        cout << endl;
    }

    // Patr贸n 2
    cout << "Patron 2:" << endl;
    int tamano = 8;
    bool** matriz2 = new bool*[tamano];
    for (int i = 0; i < tamano; i++) {
        matriz2[i] = new bool[tamano]();
    }

    for (int i = 0; i < tamano; i++) {
        matriz2[i][i] = true;
        matriz2[i][tamano - 1 - i] = true;
    }

    for (int i = 0; i < tamano; i++) {
        for (int j = 0; j < tamano; j++) {
            cout << matriz2[i][j];
        }
        cout << endl;
    }

    for (int i = 0; i < tamano; i++) {
        delete[] matriz2[i];
    }
    delete[] matriz2;

    // Patr贸n 3
    cout << "Patron 3:" << endl;
    bool matriz3[8][8];

    for (int i = 0; i < dim; i++) {
        if (i <= 1) {
            for (int j = 0; j < dim; j++) {
                if (j == 2 || j == 5) {
                    matriz3[i][j] = false;
                } else {
                    matriz3[i][j] = true;
                }
            }
        }

        if (i > 1 && i < 4) {
            for (int j = 0; j < dim; j++) {
                if (j == 0 || j == 3 || j == 6) {
                    matriz3[i][j] = false;
                } else {
                    matriz3[i][j] = true;
                }
            }
        }

        if (i > 3 && i <= 5) {
            for (int j = 0; j < dim; j++) {
                if (j == 2 || j == 5) {
                    matriz3[i][j] = false;
                } else {
                    matriz3[i][j] = true;
                }
            }
        }

        if (i >= 6) {
            for (int j = 0; j < dim; j++) {
                if (j == 0 || j == 3 || j == 6) {
                    matriz3[i][j] = false;
                } else {
                    matriz3[i][j] = true;
                }
            }
        }
    }

    for (int i = 0; i < dim; i++) {
        for (int j = 0; j < dim; j++) {
            cout << matriz3[i][j];
        }
        cout << endl;
    }

    // Patr贸n 4
    cout << "Patron 4:" << endl;
    int dim1 = 8, posicion11 = 0, posicion22 = 3;
    bool matriz4[8][8];

    for (int i = 0; i < dim1; i++) {
        if (i <= 3) {
            for (int j = 0; j < dim1; j++) {
                if (j >= posicion11 && j <= posicion22) {
                    matriz4[i][j] = true;
                } else {
                    matriz4[i][j] = false;
                }
            }
            posicion11++;
            posicion22++;
        }

        if (i == 4) {
            for (int j = 0; j < dim1; j++) {
                if (j > 2 && j <= dim1 - 2) {
                    matriz4[i][j] = true;
                } else {
                    matriz4[i][j] = false;
                }
            }
        }

        if (i > 4) {
            if (i == 5) {
                posicion11 = 2;
                posicion22 = 5;
            }

            if (i == 6) {
                posicion11 = 1;
                posicion22 = 4;
            }

            if (i == 7) {
                posicion11 = 0;
                posicion22 = 3;
            }

            for (int j = 0; j < dim1; j++) {
                if (j >= posicion11 && j <= posicion22) {
                    matriz4[i][j] = true;
                } else {
                    matriz4[i][j] = false;
                }
            }
        }
    }

    for (int i = 0; i < dim1; i++) {
        for (int j = 0; j < dim1; j++) {
            cout << matriz4[i][j];
        }
        cout << endl;
    }
}

int main() {
    dibujarPatrones();
    return 0;
}
